rules_version = '2';

// allow write: if firestore.get(

service firebase.storage {
  match /b/{bucket}/o {
    // Règles pour les documents des prospects
    match /prospects/{prospectId}/{allPaths=**} {
      // Permettre la lecture et l'écriture pour tous (formulaire public)
      allow read, write: if true;
      
      // Limiter la taille des fichiers à 10MB
      allow write: if request.resource.size < 10 * 1024 * 1024;
      
      // Limiter les types de fichiers autorisés
      allow write: if request.resource.contentType.matches('application/pdf') ||
                      request.resource.contentType.matches('image/jpeg') ||
                      request.resource.contentType.matches('image/jpg') ||
                      request.resource.contentType.matches('image/png');
    }
    
    // Règles pour les utilisateurs authentifiés (dashboard)
    match /admin/{allPaths=**} {
      allow read, write: if request.auth != null;
    }
    
    // Bloquer tout le reste
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
